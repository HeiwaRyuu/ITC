2 – Faça um programa que mostre os números na tela conforme mostrado abaixo. O usuário escolherá quantas linhas serão apresentadas (n). 

1
2 4
3 6 9
4 8 12 16
.
.
.
n

****Solução(Código):

#include <stdio.h>
#include <stdlib.h>
#include <conio.h>
int aux=0,cont;
int seq(int n){
	for(int j=1;j<=n;j++){
		aux=0;
	for(int i=0;i<j;i++){
		aux+=j;
		printf("%d ",aux);
		}
		printf("\n");
	}
}
int main(){
	do{
	system("cls");
	int linha;
	printf("\nDigite ate qual linha a cascata deve ser executada:\n\n");
	scanf("%d",&linha);
	printf("\n\n");
	seq(linha);	
	printf("\nDeseja reutilizar o programa?\nCaso sim, entre 1, caso queira sair enrte 2.\n");
	scanf("%d",&cont);
	}while(cont==1);
