2 -Faça um programa em C que recebe um número (N) digitado pelo usuário e imprime em ordem crescente os números pares de 100 até N.

Solução(Código):

#include <conio.h>
#include <stdlib.h>
#include <locale.h>

int num, sub, aux, cont;

int main(){
    do
    {
    system("cls");
    printf("\nEntre um numero maior que 100 para receber todos os numeros pares de 100 ate esse numero:\n");
    scanf("%d", &num);

    while(num<100){
    printf("\nPor favor,entre com um numero maior ou igual a 100 :\n");
    scanf("%d", &num);
    }

    if(num%2!=0){
        sub=num;
        num=num-1;
        printf("\nLista dos numeros de 100 ate %d:\n", sub);
    }
    else
        printf("\nLista dos numeros de 100 ate %d:\n", num);

    for(aux=100;aux<=num;){
         if(aux%2!=0){
            aux++;
         }
         else{
            printf("\n%d\n", aux);
            aux++;
         }
